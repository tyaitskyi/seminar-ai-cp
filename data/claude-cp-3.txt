def count_divisors(n):

    for i in range(1, int(math.sqrt(n)) + 1):

def get_factors(n):
    factors = set()
    for i in range(1, int(math.sqrt(n)) + 1):
        if n % i == 0:
            factors.add(i)
            factors.add(n // i)
    return factors

    max_count = 0

    result = 1
